73  
74              NodeList recordNodeList = root.getElementsByTagName("record");
75              for (int i = 0; i < recordNodeList.getLength(); i++) {
76                  Node recordNode = recordNodeList.item(i);

78                  NodeList recordChild = recordNode.getChildNodes();
79                  String recordstring = "";
80                  for (int j = 0; j < recordChild.getLength(); j++) {
81  
82                      Node recordChildNode = recordChild.item(j);
83  
84                      if (recordChildNode instanceof Element) {
85                          Element recordChildElement = (Element) recordChildNode;
86  
87                          String text = ((Text) recordChildElement.getFirstChild()).getData().trim();
88  
89                          recordstring = recordstring + text + "::";
90                      }
91  
92                  }
93  
94                  recordlist.add(recordstring);
95  
96              }
97  
98              System.out.println("Try to read exists xml file ");
99          } catch (ParserConfigurationException e) {
100             e.printStackTrace();
101         } catch (IOException e) {
102             e.printStackTrace();
103         } catch (SAXException e) {
104             e.printStackTrace();
105         }
106

108             if (recordlist.get(i).trim().length() > 0) {
